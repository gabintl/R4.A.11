❓ Qu'est-ce que Flutter et pourquoi?

Flutter est un framework qui crée des applications mobiles avec un seul code Dart. Normalement, créer une app Android demande Java/Kotlin et une app iOS demande Swift. Flutter unifie tout cela.

🎯 Analogie : Vous écrivez une recette une seule fois, elle marche parfaitement dans un four français ET un four américain. C'est Flutter! Les deux fours (Android/iOS) reçoivent l'app compilée exactement pareil.

Langages et outils :

  • Dart = Langage de programmation (ressemble à Java/JavaScript)
  • Flutter = Framework qui fournit les "briques" (Widgets) pour construire l'interface
  • Android Studio / VS Code = Éditeur pour écrire le code
  • Émulateur Android ou iOS = Téléphone virtuel pour voir l'app s'exécuter

⚡ État (State) - Faire Que les Données Changent

La question clé : "Cette partie de l'interface change-t-elle pendant l'utilisation?"

✅ StatelessWidget (Figé)

Utilisez si rien ne change.

Exemples :

  • Un logo de l'app
  • Titre "Accueil"
  • Un bouton normal
class Logo extends StatelessWidget {
    @override
    Widget build(BuildContext context) {
        return Image.asset(
            'assets/logo.png',
            width: 100,
        );
    }
}

⚡ StatefulWidget (Dynamique)

Utilisez si ça change (utilisateur tape, API répond, compteur augmente).

Exemples :

  • Un compteur qui augmente
  • Une liste mise à jour
  • Un formulaire avec des inputs
class Compteur extends StatefulWidget {
    @override
    _CompteurState createState() => _CompteurState();
}
class _CompteurState extends State<Compteur> {
    int count = 0;
    @override
    Widget build(BuildContext context) {
        return ElevatedButton(
            onPressed: () {
                setState(() => count++);
            },
            child: Text('Clics: \$count'),
        );
    }
}

🔑 Point Crucial : setState(() {...}) dit à Flutter "Redessine ce widget! Une variable a changé". C'est comment Flutter sait qu'il faut mettre à jour l'écran.

// Conversion vers JSON
static String _wordPairsToJson(List<WordPair> pairs) {
    return json.encode(
        pairs.map((pair) => '\(pair.first)-\(pair.second)').toList(),
    );
}

// Sauvegarde
Future<bool> _saveToStorage() async {
    var storage = await SharedPreferences.getInstance();
    return await storage.setString(
        'favorites',
        _wordPairsToJson(favorites),
    );
}

// Récupération au démarrage
Future<void> init() async {
    final storage = await SharedPreferences.getInstance();
    final data = storage.getString('favorites');

    if (data == null) {
        favorites = [];
    } else {
        favorites = _wordPairsFromJson(data);
    }
}

class _SearchPageState extends State<SearchPage> {
    String query = '';
    Future<Iterable<Location>>? locationsSearchResults;

    @override
    Widget build(BuildContext context) {
        final api = context.read<OpenWeatherMapApi>();

        return Scaffold(
            appBar: AppBar(title: const Text('Recherche')),
            body: Column(
                children: [
                    TextField(
                        onChanged: (value) => setState(() => query = value),
                    ),
                    ElevatedButton(
                        onPressed: () => setState(() {
                            locationsSearchResults = api.searchLocations(query);
                        }),
                        child: const Text('Rechercher'),
                    ),
                    Expanded(
                        child: query.isEmpty
                            ? const Text('Saisissez une ville')
                            : FutureBuilder<Iterable<Location>>(
                                future: locationsSearchResults,
                                builder: (context, snapshot) {
                                    // États du FutureBuilder
                                    if (snapshot.connectionState == ConnectionState.waiting) {
                                        return const CircularProgressIndicator();
                                    }
                                    if (snapshot.hasError) {
                                        return Text('Erreur: \(snapshot.error)');
                                    }
                                    if (!snapshot.hasData) {
                                        return const Text('Aucun résultat');
                                    }

                                    return ListView.builder(
                                        itemCount: snapshot.data!.length,
                                        itemBuilder: (context, index) {
                                            final location = snapshot.data!.elementAt(index);
                                            return ListTile(
                                                title: Text(location.name),
                                                subtitle: Text(location.country),

                                                    context,
                                                    MaterialPageRoute(

class WeatherPage extends StatefulWidget {
    final String locationName;
    final double latitude;
    final double longitude;

    @override
    State<WeatherPage> createState() => _WeatherPageState();
}

class _WeatherPageState extends State<WeatherPage> {
    late Future<Weather> _weatherFuture;

    @override
    void initState() {
        super.initState();
        final api = context.read<OpenWeatherMapApi>();
        _weatherFuture = api.getWeather(widget.latitude, widget.longitude);
    }

    @override
    Widget build(BuildContext context) {
        return FutureBuilder<Weather>(
            future: _weatherFuture,
            builder: (context, snapshot) {
                if (snapshot.connectionState == ConnectionState.waiting) {
                    return Scaffold(
                        appBar: AppBar(title: Text(widget.locationName)),
                        body: const Center(
                            child: CircularProgressIndicator(),
                        ),
                    );
                }

class AuthState extends ChangeNotifier {
    User? _currentUser;
    String? _token;
    String error = '';

    User? get currentUser => _currentUser;
    String? get token => _token;
    bool get isLoggedIn => _currentUser != null;

    // Connexion
    Future<Result<User, String>> login(
        String username,
        String password,
    ) async {
        try {
            final loginResponse = await http.post(
                Uri.parse('\(Configs.baseUrl)/auth/login'),
                headers: {HttpHeaders.contentTypeHeader: 'application/json'},
                body: json.encode({'username': username, 'password': password}),
            );

            if (loginResponse.statusCode == HttpStatus.ok) {
                // Récupérer le token
                _token = json.decode(loginResponse.body)['token'];

                // Récupérer les infos utilisateur
                final userResponse = await http.get(
                    Uri.parse('\(Configs.baseUrl)/users/me'),
                    headers: {
                        HttpHeaders.authorizationHeader: 'Bearer \(_token)',
                        HttpHeaders.contentTypeHeader: 'application/json',
                    },
                );

                if (userResponse.statusCode == HttpStatus.ok) {
                    _currentUser = User.fromJson(json.decode(userResponse.body));
                    notifyListeners();
                    return Result.success(_currentUser!);
                }
            } else {
                error = loginResponse.statusCode == HttpStatus.badRequest ? 
                    'Identifiant ou mot de passe incorrect' : 
                    'Une erreur est survenue';
            }
        } catch (e) {
            error = e.toString();
        }

        logout();
        return Result.failure(error);
    }

    // Inscription
    Future<Result<User, String>> signup(
        String username,
        String password,
    ) async {
        try {
            final response = await http.post(
                Uri.parse('\(Configs.baseUrl)/auth/register'),
                headers: {HttpHeaders.contentTypeHeader: 'application/json'},
                body: json.encode({'username': username, 'password': password}),
            );

            if (response.statusCode == HttpStatus.created) {
                // Connexion automatique après inscription
                return await login(username, password);
            }
            error = 'Erreur lors de l\'inscription';
        } catch (e) {
            error = e.toString();
        }

        return Result.failure(error);
    }

    void logout() {
        _currentUser = null;
        _token = null;
        notifyListeners();
    }
}

📝 Pattern Result<Success, Error>

Fichier: lib/result.dart (fourni)

class Result<S, E> {
    const Result._({this.success, this.error});

    final S? success;
    final E? error;

    factory Result.success(S value) => Result._(success: value);
    factory Result.failure(E value) => Result._(error: value);

    R when<R>({
        required R Function(S) success,
        required R Function(E) failure,
    }) {
        if (this.success != null) {
            return success(this.success as S);
        } else {
            return failure(this.error as E);
        }
    }
}

Future<Result<String, String>> updatePollImage(
    int id,
    File imageFile,

    final fileContent = await imageFile.readAsBytes();

    final request = http.MultipartRequest(
        'POST',

    )
        ..headers.addAll({

        await request.send(),

        notifyListeners();
        return Result.success('Image uploadée');
    }
    return Result.failure('Erreur upload');
}

class UserService {
    static const baseUrl = 'https://api.example.com';

    // Récupérer tous les utilisateurs
    Future<List<User>> getUsers() async {
        final response = await http.get(
            Uri.parse('\(baseUrl)/users'),
        );
        if (response.statusCode == 200) {
            final List data = json.decode(response.body);
            return data.map((u) => User.fromJson(u)).toList();
        }
        throw Exception('Erreur');
    }

    // Créer un utilisateur
    Future<User> createUser(User user) async {
        final response = await http.post(
            Uri.parse('\(baseUrl)/users'),
            body: json.encode(user.toJson()),
        );
        if (response.statusCode == 201) {
            return User.fromJson(json.decode(response.body));
        }
        throw Exception('Erreur création');
    }
}

🛡️ Pattern Result (Gestion d'Erreurs)

Fichier: lib/models/result.dart

class Result<S, E> {
    final S? success;
    final E? error;

    Result._({this.success, this.error});

    factory Result.ok(S value) => Result._(success: value);
    factory Result.err(E error) => Result._(error: error);

    bool get isSuccess => success != null;
    bool get isError => error != null;

    R when<R>({
        required R Function(S) ok,
        required R Function(E) err,
    }) {
        if (isSuccess) {
            return ok(success as S);
        } else {
            return err(error as E);
        }
    }
}

// Utilisation
final result = await api.fetchData();
result.when(
    ok: (data) => print('Succès: ' + data.toString()),
    err: (err) => print('Erreur: ' + err.toString()),
);
